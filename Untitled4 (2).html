# Total Spend (Monetary Value) 
total_spend = df.groupby('CustomerID')['TotalAmount'].sum().rename('Total_Spend')

#  Purchase Frequency (Number of Invoices) 
purchase_frequency = df.groupby('CustomerID')['InvoiceNo'].nunique().rename('Purchase_Frequency')

#  Average Order Value (Monetary per Invoice) 
avg_order_value = (total_spend / purchase_frequency).rename('Avg_Order_Value')
In[6]:
# Weekend Purchase Ratio
df['IsWeekend'] = df['InvoiceDate'].dt.dayofweek.isin([5, 6]).astype(int)
weekend_ratio = df.groupby('CustomerID')['IsWeekend'].mean().rename('Weekend_Purchase_Ratio')

# Cyclical Time Features (Month & Day-of-Week using sine and cosine) 
df['Month'] = df['InvoiceDate'].dt.month
df['DayOfWeek'] = df['InvoiceDate'].dt.dayofweek

df['Month_Sin'] = np.sin(2 * np.pi * df['Month'] / 12)
df['Month_Cos'] = np.cos(2 * np.pi * df['Month'] / 12)

df['DOW_Sin'] = np.sin(2 * np.pi * df['DayOfWeek'] / 7)
df['DOW_Cos'] = np.cos(2 * np.pi * df['DayOfWeek'] / 7)
# Aggregate cyclical features at customer level 
cyclical_features = df.groupby('CustomerID')[['Month_Sin', 'Month_Cos', 'DOW_Sin', 'DOW_Cos']].mean()
In[8]:

).reset_index()
